replaceAll :: Eq a => Replace a Source

Replace all occurrences.

>>> replaceAll "br" "BR" "abracadabra"
"aBRacadaBRa"

>>> replaceAll "ba" "BA" "abracadabra"
"abracadabra"

>>> replaceAll "" "BR" "abracadabra"
"abracadabra"
