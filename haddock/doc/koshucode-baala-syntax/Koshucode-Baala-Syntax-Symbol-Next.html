nextSpace :: Textual t => Next t Int Source

Get next spaces. Space character is decided by isSpace.

>>> nextSpace " abc"
("abc", 1)

>>> nextSpace " \tabc"
("abc", 2)

nextQQ :: Textual t => AbNext t t Source

Get next double quoted text. A single double quote ends text. A double double quote does not end text, but it is converted to double quote.

>>> nextQQ "abc\" def"
Right (" def","abc")

>>> nextQQ "abc\"\"def\" ghi"
Right (" ghi","abc\"def")

nextBefore :: Textual t => t -> AbNext t t Source

Get next text before given delimiter.

>>> nextBefore "''" "abc'' def"
Right (" def","abc")
