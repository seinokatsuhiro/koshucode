type PoScale a = Map a ([a], Maybe Int) Source

Mapping from element to subelements and scale. Scale of element is defined by 1 + the maximum scale of subelements. If elements are cyclic, scales of these elements are -1.

type PoScaleCalc a = [(a, a)] -> [(a, Int)] Source

Calculate partial-order scale.

poScale :: PoScale a -> [(a, Int)] Source

Extract partial-order scale.

poScaleHeight :: Ord a => PoScaleCalc a Source

Calculate upward partial-order scale.

>>> poScaleHeight [("a","b"), ("a","c"), ("b","d"), ("c","d")]
[("a",0), ("b",1), ("c",1), ("d",2)]

poScaleDepth :: Ord a => PoScaleCalc a Source

Calculate downward partial-order scale.

>>> poScaleDepth [("a","b"), ("a","c"), ("b","d"), ("c","d")]
[("a",2), ("b",1), ("c",1), ("d",0)]
