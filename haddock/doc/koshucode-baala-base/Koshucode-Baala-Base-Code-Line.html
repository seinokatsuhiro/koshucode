linesCrlf :: String -> [String] Source

Split string into lines. The result strings do not contain carriage returns (\r) and line feeds (\n).

>>> linesCrlf "aaa\nbbb\r\nccc\n\nddd\n"
["aaa","bbb","ccc","","ddd"]
