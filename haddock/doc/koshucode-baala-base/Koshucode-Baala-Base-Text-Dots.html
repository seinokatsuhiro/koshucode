dotsStrings :: String -> [String] Source

Divide dots-pattern string.

>>> dotsStrings "abc ... xyz"
["abc","...","xyz"]

>>> dotsStrings "a ... h ... ... v ..."
["a","...","h","...","v","..."]

undotsStrings :: [String] -> String Source

Concatenate dots-pattern strings, inserting one space between dots and word.

>>> undotsStrings ["abc", "...", "xyz"]
"abc ... xyz"

>>> undotsStrings ["a", "...", "h", "...", "...", "v", "..."]
"a ... h ... v ..."

takeDots :: [String] -> String -> Maybe [String] Source

Take dots-parts from string.

>>> takeDots ["a", "...", "h", "...", "v", "..."] "abcdefghijklmnopqrstuvwxyz"
Just ["bcdefg","ijklmnopqrstu","wxyz"]

>>> takeDots ["a", "...", "h", "...", "v", "..."] "abcdefg"
Nothing
