Reexport from Data.Char

isSpace :: Char -> Bool

Returns True for any Unicode space character, and the control characters \t, \n, \r, \f, \v.

Trim

trimBegin :: Textual t => t -> t Source

Remove space and space-like characters from the beginning of string.

>>> trimBegin "  abc  "
"abc  "

trimEnd :: Textual t => t -> t Source

Remove space and space-like characters from the end of string.

>>> trimEnd "  abc  "
"  abc"

trimBoth :: Textual t => t -> t Source

Remove space and space-like characters from the beginning and end of string.

>>> trimBoth "  abc  "
"abc"

Padding

padBegin :: Int -> StringMap Source

Add spaces to the left.

>>> padBegin 10 "abc"
"       abc"

padEnd :: Int -> StringMap Source

Add spaces to the right.

>>> padEnd 10 "abc"
"abc       "

padBeginWith :: Char -> Int -> StringMap Source

Add given character to the left.

>>> padBeginWith '.' 10 "abc"
".......abc"

padEndWith :: Char -> Int -> StringMap Source

Add given character to the right.

stringWidth :: String -> Int Source

Calculate width of string.

addSpace :: Textual t => t -> t Source

Add space character if first character is non-space.

>>> addSpace "aaa"
" aaa"

>>> addSpace " bbb"
" bbb"

>>> addSpace ""
""
