Infix operators

(&) :: a -> b -> (a, b) infixr 0 Source

Pairing operator.

>>> "a" & "b"
("a", "b")

>>> "a" & "b" & "c"
("a", ("b", "c"))

(/$/) :: Map [a] -> Map [a] Source

Apply function to reversed list, and reverse back.

>>> take 3 $ "abcdefg"
"abc"

>>> take 3 /$/ "abcdefg"
"efg"

(++) :: Monoid a => a -> a -> a infixr 6 Source

Associative operator of monoid, same as mappend or infix <>.

>>> Just "abc" ++ Nothing ++ Just "def"
Just "abcdef"

(<++>) :: (a -> [b]) -> [a] -> [b] Source

Same as concatMap.

>>> words <++> ["foo bar", "baz quux"]
["foo","bar","baz","quux"]

(<?>) :: (a -> Maybe b) -> [a] -> [b] Source

Same as mapMaybe.

(<$$>) :: (Functor f, Functor g) => (a -> b) -> f (g a) -> f (g b) infixl 4 Source

Double fmap.

>>> length <$$> [("a", "apple"), ("b", "banana"), ("c", "cocoa")]
[("a",5), ("b",6), ("c",5)]

(<#>) :: (Monad m, Traversable t) => (a -> m b) -> t a -> m (t b) Source

Monadic mapping, same as mapM.

(<#!>) :: (Monad m, Traversable t) => (a -> m b) -> t a -> m () Source

Execution by monadic mapping, same as mapM_.
