Infix operators

(&) :: a -> b -> (a, b) infixr 0 Source

Pairing operator.

>>> "a" & "b"
("a", "b")

>>> "a" & "b" & "c"
("a", ("b", "c"))

(++) :: Monoid a => a -> a -> a infixr 6 Source

Associative operator of monoid, same as mappend or infix <>.

>>> Just "abc" ++ Nothing ++ Just "def"
Just "abcdef"

(<++>) :: (a -> [b]) -> [a] -> [b] Source

Same as concatMap.

>>> words <++> ["foo bar", "baz quux"]
["foo","bar","baz","quux"]

(<$$>) :: (Functor f, Functor g) => (a -> b) -> f (g a) -> f (g b) infixl 4 Source

Double fmap.

>>> length <$$> [("a", "apple"), ("b", "banana"), ("c", "cocoa")]
[("a",5), ("b",6), ("c",5)]
