Presend & new terms

preTerms :: TermPicker c -> [TermName] Source

List of present terms. The first argument of termPicker is treated as target terms, and the second is terms of input heading. Because firstly target terms are given to relmap operator, secondly input heading is fixed.

>>> preTerms $ termPicker "/a /b /c" "/a /b /d /e"
[TermName EQ "a",TermName EQ "b"]

newTerms :: TermPicker c -> [TermName] Source

List of new terms.

>>> newTerms $ termPicker "/a /b /c" "/a /b /d /e"
[TermName EQ "c"]

preTermsExist :: Test (TermPicker c) Source

Test present terms exist.

>>> preTermsExist $ termPicker "/d /e" "/a /b /c"
False

>>> preTermsExist $ termPicker "/c /d /e" "/a /b /c"
True

newTermsExist :: Test (TermPicker c) Source

Test new terms exist.

>>> newTermsExist $ termPicker "/a /b /c" "/a /b /d /e"
True

>>> newTermsExist $ termPicker "/a /b" "/a /b /d /e"
False

pickTerms :: TermPick c Source

Pick target terms from input terms.

>>> pickTerms (termPicker "/b /d" "/a /b /c /d") "ABCD"

cutTerms :: TermPick c Source

backwardTerms :: TermPick c Source

towardTerms :: Bool -> TermPick c Source
