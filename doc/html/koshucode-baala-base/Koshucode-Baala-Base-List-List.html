Take

takeFirst :: [a] -> [a] Source

Take first element; if empty list is given, returns empty list.

>>> takeFirst "abc"
"a"

>>> takeFirst ""
""

takeLast :: [a] -> [a] Source

Take last element; if empty list is given, returns empty list.

>>> takeLast "abc"
"c"

takeOdd :: [a] -> [a] Source

Take elements at odd positions, i.e., first, third, ...

>>> takeOdd "abcdeft"
"acet"

takeEven :: [a] -> [a] Source

Take elements at even positions. i.e., second, fourth, ...

>>> takeEven "abcdeft"
"bdf"

takeFill :: a -> Int -> [a] -> [a] Source

Take N elements with filler.

>>> takeFill 0 3 [0 .. 8]
  [0,1,2]

>>> takeFill 0 5 [6 .. 8]
  [6,7,8,0,0]

takeTailFill :: a -> Int -> [a] -> [a] Source

Take tail-side N elements with filler.

>>> takeTailFill 0 3 [0 .. 8]
  [6,7,8]

>>> takeTailFill 0 5 [6 .. 8]
  [0,0,6,7,8]

setList :: Ord a => [a] -> [a] Source

Convert to set-like list, in other words, remove duplicate elements and sort list.

setEq :: Ord a => [a] -> [a] -> Bool Source

Set-like equality, in other words, duplication and ordering are ignored.

Examples

>>> duplicates "banana"
"ana"

>>> unique "banana"
"ban"

>>> unionUp "cde" "abc"
"deabc"

>>> List.union "cde" "abc"
"cdeab"
