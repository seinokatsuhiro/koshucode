divideBy :: (a -> Bool) -> [a] -> [[a]] Source

Divide list.

Examples 

>>> splitBy (== '|') "b c"
Left "b c"

>>> splitBy (== '|') "a | b | c"
Right ("a ", '|', " b | c")

>>> divide '|' "a|bb||ccc|"
["a", "bb", "", "ccc", ""]

>>> divideBy (== '|') "a|bb||ccc|"
["a", "bb", "", "ccc", ""]
